Features

Installation

Node.js 12.0.0 or newer is required

NPM

npm i api-devup --save

Usage example

const API = require("api-devup");

const api = new API({
  key: process.env.KEY,
});

async function run() {
  const response = await api.getGroups({
    user_id: 1,
  });
  console.log(response);
}

run().catch(console.log);
